How To Use?

# Add AnyCable to your Gemfile

gem 'anycable', group: :production


# Generate server script

rails generate anycable


# Run gRPC server

./bin/anycable


# Run WebSocket server (e.g. anycable-go)

anycable-go -addr=0.0.0.0:3334
